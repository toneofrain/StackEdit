import java.util.Scanner;
  
public class Main {
  public static void main(String[] args){
    Scanner in = new Scanner(System.in);
    String str = new in.nextLine();
    StrigBuilder sb = new StringBuilder();
    
   	for (char c : str.toCharArray()) {
		if (c >= 'a')
          sb.append((char) (c - 32));
      	else
          sb.append((char) (c + 32));
    }
    System.out.print(sb);
  }
}
